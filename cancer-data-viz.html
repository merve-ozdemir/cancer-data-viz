#Plot 1: Distribution of Cancer Types We start with a bar plot showing the frequency of different cancer types.

# Bar plot
plot_cancer_type <- ggplot(data, aes(x = Cancer_Type)) +
  geom_bar(fill = "skyblue") +
  coord_cartesian(ylim = c(425, 525)) +
  labs(
    title = "Cancer Type Distribution",
    x = "Cancer Type",
    y = "Count"
  ) +
  theme_minimal()

# Display the plot
plot_cancer_type

#Plot 2: Age Distribution of Patients Next, we visualize the distribution of patient ages using a density plot. Transparency (alpha) is added to make the background and gridlines more visible.

plot_age <- ggplot(data, aes(x = Age)) +
  geom_density(fill = "skyblue", color = "darkblue", alpha = 0.4) +
  scale_x_continuous(limits = c(0, 100), breaks = seq(0, 100, 20)) +
  labs(
    title = "Age Distribution of Cancer Patients",
    x = "Age",
    y = "Density"
  ) +
  theme_minimal()

plot_age
